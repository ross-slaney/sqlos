## AuthServer Services

### SqlOSAuthService

Core authentication service for login, session management, and token operations.

```csharp
// Password login
Task<SqlOSLoginResult> LoginWithPasswordAsync(
    SqlOSPasswordLoginRequest request, HttpContext httpContext, CancellationToken ct);

// Signup
Task<SqlOSLoginResult> SignUpAsync(
    SqlOSSignupRequest request, HttpContext httpContext, CancellationToken ct);

// Organization selection (multi-org)
Task<SqlOSTokenResponse> SelectOrganizationAsync(
    SqlOSSelectOrganizationRequest request, HttpContext httpContext, CancellationToken ct);

// Code exchange (SSO/OIDC)
Task<SqlOSTokenResponse> ExchangeCodeAsync(
    SqlOSExchangeCodeRequest request, HttpContext httpContext, CancellationToken ct);

// Refresh token
Task<SqlOSTokenResponse> RefreshAsync(SqlOSRefreshRequest request, CancellationToken ct);

// Logout
Task LogoutAsync(string? refreshToken, string? sessionId, CancellationToken ct);
Task LogoutAllAsync(string userId, CancellationToken ct);

// Token validation
Task<SqlOSValidatedToken?> ValidateAccessTokenAsync(string rawToken, CancellationToken ct);

// Session creation (for OIDC complete flows)
Task<SqlOSTokenResponse> CreateSessionTokensForUserAsync(
    SqlOSUser user, SqlOSClientApplication client, string? organizationId,
    string authenticationMethod, string? userAgent, string? ipAddress, CancellationToken ct);

// Password reset
Task<string> CreatePasswordResetTokenAsync(SqlOSForgotPasswordRequest request, CancellationToken ct);
Task ResetPasswordAsync(SqlOSResetPasswordRequest request, CancellationToken ct);

// Email verification
Task<string> CreateEmailVerificationTokenAsync(SqlOSCreateVerificationTokenRequest request, CancellationToken ct);
Task VerifyEmailAsync(SqlOSVerifyEmailRequest request, CancellationToken ct);
```
